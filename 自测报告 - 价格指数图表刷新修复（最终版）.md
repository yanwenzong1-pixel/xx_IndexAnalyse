## 附录：测试脚本

### 自动化测试脚本（Chrome 控制台）

```javascript
// 自动化 F5 刷新测试
(async function autoRefreshTest() {
  const maxAttempts = 50;
  let successCount = 0;
  
  for (let i = 0; i < maxAttempts; i++) {
    try {
      // 等待页面加载
      await new Promise(resolve => setTimeout(resolve, 2000));
      
      // 检查图表是否存在
      const chartElement = document.querySelector('.echarts');
      if (chartElement && chartElement.offsetWidth > 0) {
        successCount++;
        console.log(`刷新 ${i + 1}/${maxAttempts}: ✅ 成功`);
      } else {
        console.log(`刷新 ${i + 1}/${maxAttempts}: ❌ 失败`);
      }
      
      // 刷新页面（除了最后一次）
      if (i < maxAttempts - 1) {
        location.reload();
        await new Promise(resolve => setTimeout(resolve, 1000));
      }
    } catch (error) {
      console.error(`刷新 ${i + 1}: 错误`, error);
    }
  }
  
  console.log(`\n=== 测试结果 ===`);
  console.log(`总次数：${maxAttempts}`);
  console.log(`成功：${successCount}`);
  console.log(`失败：${maxAttempts - successCount}`);
  console.log(`成功率：${(successCount / maxAttempts * 100).toFixed(2)}%`);
})();
```
