### 步骤 2：修改 TabChart 组件

找到 `TabChart` 组件定义（约第 1539 行），将整个组件替换为以下代码：

```javascript
// TabChart 组件 - 价格指数、成交量、风险趋势切换
const TabChart = ({ data }) => {
  const [activeTab, setActiveTab] = React.useState('price');
  const [riskData, setRiskData] = React.useState(null);
  const [riskLoading, setRiskLoading] = React.useState(false);

  // 获取风险数据
  React.useEffect(() => {
    if (activeTab === 'risk' && !riskData) {
      setRiskLoading(true);
      
      fetch('http://localhost:5000/api/risk/history?days=252')
        .then(res => res.json())
        .then(result => {
          if (result.success) {
            setRiskData(result.data);
            // 缓存到 localStorage
            localStorage.setItem('riskHistoryData', JSON.stringify(result.data));
            localStorage.setItem('riskHistoryTimestamp', Date.now().toString());
          }
          setRiskLoading(false);
        })
        .catch(err => {
          console.error('风险数据获取失败:', err);
          setRiskLoading(false);
        });
    }
  }, [activeTab]);
  
  return (
    <div className="bg-white dark:bg-gray-800 rounded-xl shadow-lg p-6 mb-8 custom-card">
      {/* Tab 标签栏 */}
      <div className="flex border-b border-gray-200 dark:border-gray-700 mb-6">
        <button
          className={`px-6 py-3 font-medium transition-all duration-300 relative ${
            activeTab === 'price' ? 'text-blue-600 dark:text-blue-400' : 'text-gray-600 dark:text-gray-400 hover:text-blue-600 dark:hover:text-blue-400'
          }`}
          onClick={() => setActiveTab('price')}
        >
          价格指数
          {activeTab === 'price' && (
            <span className="absolute bottom-0 left-0 right-0 h-0.5 bg-gradient-to-r from-blue-500 to-blue-600 rounded-full"></span>
          )}
        </button>
        <button
          className={`px-6 py-3 font-medium transition-all duration-300 relative ${
            activeTab === 'volume' ? 'text-blue-600 dark:text-blue-400' : 'text-gray-600 dark:text-gray-400 hover:text-blue-600 dark:hover:text-blue-400'
          }`}
          onClick={() => setActiveTab('volume')}
        >
          成交量
          {activeTab === 'volume' && (
            <span className="absolute bottom-0 left-0 right-0 h-0.5 bg-gradient-to-r from-blue-500 to-blue-600 rounded-full"></span>
          )}
        </button>
        <button
          className={`px-6 py-3 font-medium transition-all duration-300 relative ${
            activeTab === 'risk' ? 'text-blue-600 dark:text-blue-400' : 'text-gray-600 dark:text-gray-400 hover:text-blue-600 dark:hover:text-blue-400'
          }`}
          onClick={() => setActiveTab('risk')}
        >
          风险趋势
          {activeTab === 'risk' && (
            <span className="absolute bottom-0 left-0 right-0 h-0.5 bg-gradient-to-r from-blue-500 to-blue-600 rounded-full"></span>
          )}
        </button>
      </div>
      
      {/* 图表内容 */}
      <div className="h-[400px] rounded-lg overflow-hidden shadow-inner bg-gray-50 dark:bg-gray-900/50">
        {activeTab === 'price' ? (
          <PriceChart key={`price-${data?.length || 0}`} data={data} />
        ) : activeTab === 'volume' ? (
          <TurnoverChart key={`volume-${data?.length || 0}`} data={data} />
        ) : riskLoading ? (
          <div className="flex items-center justify-center h-full">
            <div className="text-gray-500">加载中...</div>
          </div>
        ) : riskData ? (
          <RiskTrendChart data={riskData} />
        ) : (
          <div className="flex items-center justify-center h-full">
            <div className="text-gray-500">数据加载失败</div>
          </div>
        )}
      </div>
    </div>
  );
};
```
